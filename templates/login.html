@app.route('/login', methods=['GET', 'POST']) def login(): if request.method == 'POST': username = request.form['username'] password = request.form['password'] if username not in users: return "User not found ❌" if users[username]['password'] != password: return "Wrong password ❌" session['username'] = username # 🔥 save user return redirect(url_for('dashboard')) return render_template('login.html') conn = sqlite3.connect('users.db') c = conn.cursor() c.execute("SELECT * FROM users WHERE username=? AND password=?", (username, password)) user = c.fetchone() conn.close() if user: session['username'] = username return redirect('/dashboard') else: return "Invalid login ❌"
